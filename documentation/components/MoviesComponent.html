import { Component } from '@angular/core';
import { RouterOutlet, RouterModule, ActivatedRoute, Router} from '@angular/router';
import { Dataservice } from './data.service';
import { webService } from './web.service';
import { FormBuilder, ReactiveFormsModule, Validators } from '@angular/forms';
import { CommonModule } from '@angular/common';
import { AuthService } from '@auth0/auth0-angular';

/**
 * outlining decorators that provide config metadata to determine how the component is processed
 */
@Component({
  selector: 'movies',
  standalone: true,
  imports: [RouterOutlet, RouterModule, CommonModule, ReactiveFormsModule],
  providers: [Dataservice, webService],
  templateUrl: './movies.component.html',
  styleUrl: './movies.component.css'
})

/**
 * defining the class for the movies component
 */
export class MoviesComponent {
  /**
   * variable for storing movies
   */
  movie_list: any;

  /**
   * variable for pagination
   */
  page: number = 1;

  /**
   * variable for adding a new movie form
   */
  movieForm: any;

  /**
   * declaring variable for adding elements to a new movie
   */
  movie: any = {
    title: '',
    director: '',
    genre: ''
  };

  /**
   * the constructor for the Web Service and data service
   * @param dataService using the data service to share data across the various components
   * @param webService using the web service to handle HTTP requests
   * @param route shows details for movies and dynamic routes
   * @param formBuilder service used to create forms for the reviews
   * @param authService manages user authentication
   */
  constructor(public dataService: Dataservice,
              private webService: webService,
              private route: ActivatedRoute,
              private router: Router,
              private formBuilder: FormBuilder,
              public authService: AuthService) {}
/**
 * utilised to load movies from the component and set current page
 */
  ngOnInit() {
    this.movieForm = this.formBuilder.group( {
      title: ["", Validators.required],
      director: ["", Validators.required],
      genre: ["", Validators.required]
    });
    /**
     * checks for page value and assigns value to the page and stores it in the session storage
     */
    if (sessionStorage['page']) {
      this.page = Number(sessionStorage['page']);
    }
    //this.movies_list = this.dataService.getMovies(this.page);
    this.webService.getMovies(this.page).subscribe( (response) => { this.movie_list = response; })
  }

  onSubmit() {
    this.webService.addMovie(this.movieForm.value).subscribe(
      (response) => {
        console.log('Movie added successfully:', response);
        // Redirect to the movie list or any other page after success
        this.router.navigate(['/movies']);
      },
      (error) => {
        console.error('Error adding movie:', error);
        alert('There was an error adding the movie.');
      }
    );
  }

  /**
   *implementing pagination for managing the navigation of pages between movies
   */
  previousPage() {
    if (this.page > 1) {
        this.page = this.page - 1;
        sessionStorage['page'] = this.page;
        this.webService.getMovies(this.page).subscribe( (response) => { this.movie_list = response; })
    }
  }

  /**
   *implementing pagination for the method, checks the current page is less than the next possible page
   */
  nextPage() {
    if (this.page < this.dataService.getLastPageNumber())  {
        this.page = this.page + 1;
        sessionStorage['page'] = this.page;
        this.webService.getMovies(this.page).subscribe( (response) => { this.movie_list = response; })
    }
  }

  /**
   * method for checking movie data interaction
   * @param control checks whether the data is invalid or the movie section has a missing field
   * @returns warning to the user of missing field data
   */
  isInvalid(control: any) {
    return this.movieForm.controls[control].invalid && this.movieForm.controls[control].touched;
  }

  /**
 * method to check whether the username or comment section has been interacted with
 * @returns warning to the user of missing critical data
 */
  isUntouched() {
    return this.movieForm.controls.username.pristine ||
           this.movieForm.controls.comment.pristine;
  }

  /**
   * method to check whether a username or comment may be invalid
   * @returns warning to the user of invalid or missing data
   */
  isIncomplete() {
    return this.isInvalid('title') ||
           this.isInvalid('director') ||
           this.isInvalid('genre') ||
           this.isUntouched();
  }

 }
