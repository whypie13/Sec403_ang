import { Component } from '@angular/core';
import { RouterOutlet, ActivatedRoute, Router } from '@angular/router';
import { Dataservice } from './data.service';
import { CommonModule } from '@angular/common';
import { ReactiveFormsModule } from '@angular/forms';
import { FormBuilder, Validators } from '@angular/forms';
import { AuthService } from '@auth0/auth0-angular';
import { webService } from './web.service';

/**
 * defining a set of components for movie
 */
@Component({
  selector: 'movie',
  imports: [RouterOutlet, CommonModule, ReactiveFormsModule],
  providers: [Dataservice, webService],
  templateUrl: './movie.component.html',
  styleUrl: './movie.component.css'
})


export class movieComponent {

  /**
   * holds the list of movies displayed within the component
   */
  movie_list: any;

  /**
   * provides the description for the movies in the application
   */
  loremIpsum: any;

  /**
   * form allowing users to submit movie reviews
   */
  reviewForm: any;

  /**
   * stores all reviews assigned to the movies
   */
  review_list: any;

  /**
   * property used to identify a specific movie by a unique ID
   */
  movieID: any;

  /**
   * property used to identify a specific review by a unique ID
   */
  reviewID: any;

  /**
   * the constructors imported for the use of the movie component
   * @param dataService using the data service to share data across the various components
   * @param webService using the web service to handle HTTP requests
   * @param route shows details for movies and dynamic routes
   * @param formBuilder service used to create forms for the reviews
   * @param authService manages user authentication
   */
  constructor( public dataService: Dataservice,
               private webService: webService,
               private route: ActivatedRoute,
               private formBuilder: FormBuilder,
               public authService: AuthService,
               private router: Router) {}


  /**
   * setting up controls for the review form data
   */
  ngOnInit() {
    this.reviewForm = this.formBuilder.group( {
      username: ["", Validators.required],
      comment: ["", Validators.required],
      stars: 5
    });
    this.movieID = this.route.snapshot.paramMap.get('id');
    this.webService.getMovie(this.route.snapshot.paramMap.get('id'))
    .subscribe( (response) => {
      this.movie_list = [response];

        this.dataService.getLoremIpsum(1)
      .subscribe( (response: any) => {
        this.loremIpsum = response.text.slice(0,300)
      });
    });

    this.reviewID = this.webService.getReviews(this.route.snapshot.paramMap.get('id'))
    this.webService.getReviews(this.route.snapshot.paramMap.get('id'))
    .subscribe( (response) => {
      this.review_list = response;
    });
  }

  /**
   * method that allows a selected movie to be deleted from the movies collection
   */
  deleteMovie(): void {
    if (this.movieID && confirm('Are you sure you want to delete this movie?')) {
      this.webService.deleteMovie(this.movieID).subscribe(
        () => {
          console.log('Movie deleted successfully');
          // After deletion, navigate back to the movie list or homepage
          this.router.navigate(['/movies']);  // Adjust this route based on your application structure
        },
        (error) => {
          console.error('Error deleting movie:', error);
          alert('There was an error deleting the movie.');
        }
      );
    }
  }

  /**
   * method that allows a selected review to be deleted from the movie collection
   * @param reviewID uses the ID object to identify a specific review
   */
  deleteReview(reviewID:string): void {
    if (this.reviewID && confirm('Are you sure you want to delete this review?')) {
      this.webService.deleteReview(this.movieID, reviewID).subscribe(
        () => {
          console.log('Review deleted successfully');
          window.location.reload();
        },
        (error) => {
          console.error('Error deleting review:', error);
          alert('There was an error deleting the review.');
        }
      );
    }
  }

  /**
   * setting up the method to handle the submission of review data for a movie
   */
  onSubmit() {
    this.webService.postReview(
        this.route.snapshot.paramMap.get('id'),
        this.reviewForm.value)
        .subscribe( (response) => {
          this.reviewForm.reset();
          this.webService.getReviews(this.route.snapshot.paramMap.get('id'))
            .subscribe( (response) => {
              this.review_list = response;
          });
        });
  }

  /**
   * method for checking review data interaction
   * @param control checks whether the data is invalid or the review section has a missing field
   * @returns warning to the user of missing field data
   */
  isInvalid(control: any) {
    return this.reviewForm.controls[control].invalid && this.reviewForm.controls[control].touched;
  }


/**
 * method to check whether the username or comment section has been interacted with
 * @returns warning to the user of missing critical data
 */
  isUntouched() {
    return this.reviewForm.controls.username.pristine ||
           this.reviewForm.controls.comment.pristine;
  }

  /**
   * method to check whether a username or comment may be invalid
   * @returns warning to the user of invalid or missing data
   */
  isIncomplete() {
    return this.isInvalid('username') ||
           this.isInvalid('comment') ||
           this.isUntouched();
  }
}
