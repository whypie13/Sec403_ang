import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';

/**
 * The Web Service provides access to the endpoints of the Movie Directory API
 */

@Injectable()
export class webService {

  /**
   * The default page size to be returned
   */
  pageSize: number = 4;

  /**
   * The constructor for the Web Service
   * @param http Injecting the HttpClient to the WebService class
   */
  constructor(private http: HttpClient) {}

/**
 * Fetch a page of movies from the Movie Directory API
 * @param page The page number requested
 * @returns An Observable for the collection of movies
 */

  getMovies(page: number) {
    return this.http.get<any>(
      'http://127.0.0.1:5000/api/v1.0/movies?pn=' + page + '&ps=' + this.pageSize);
  }

/**
 * Fetch a single movie from the Movie Directory API
 * @param id The identiier of the movie requested
 * @returns An Observable for the movie object
 */
  getMovie( id: any){
    return this.http.get<any>(
      'http://127.0.0.1:5000/api/v1.0/movies/' + id);
  }

  /**
   * method that uses a form to submit a new movie to the movies collection
   * @param movie provides the new movie with a unique ID
   * @returns a new movie added to the movies collection
   */
  addMovie(movie:any): Observable<any>{
    let postData = new FormData();
    postData.append("title", movie.title);
    postData.append("director", movie.director);
    postData.append("genre", movie.genre);
    return this.http.post<any>(
      'http://127.0.0.1:5000/api/v1.0/movies', postData);
  }

  /**
   * Fetch the collection of reviews for a specific movie from the Movie Directory API
   * @param id The identiier of the movie for which the reviews are requested
   * @returns An Observable for the collection of reviews
   */
  getReviews( id: any){
    return this.http.get<any>(
      'http://127.0.0.1:5000/api/v1.0/movies/' + id + '/reviews');
  }

  /**
   * Add a review to the collection of the specified movie
   * @param id The identifier of the movie for which the review is provided
   * @param review The review to be added
   * @returns An Obserable on an object containing a link to the new review
   */
  postReview(id: any, review: any) {
    let postData = new FormData();
    postData.append("username", review.username);
    postData.append("comment", review.comment);
    postData.append("stars", review.stars);
    return this.http.post<any> (
      'http://127.0.0.1:5000/api/v1.0/movies/' + id + '/reviews', postData);
  }

  /**
   * method to delete a movie from the movies collection
   * @param id uses a unique ID to identify the movie in question
   * @returns deletion of the movie from the collection
   */
  deleteMovie( id: any){
    return this.http.delete<any>(
      'http://127.0.0.1:5000/api/v1.0/movies/' + id);
  }

  /**
   * method to delete a review from a specific movie
   * @param id uses a unique ID to identify the movie in question
   * @param reviewID uses a unique ID to identify a single review for that movie
   * @returns deletion of the review from that movie
   */
  deleteReview( id: any, reviewID:any){
    return this.http.delete<any>(
      'http://127.0.0.1:5000/api/v1.0/movies/' + id + '/reviews/' + reviewID);
  }
}
