import { Observable } from 'rxjs';
import jsonData from '../assets/moviecollection.json';
import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';

/**
 * The data Service provides access to the endpoints of the Movie Directory API
 */
@Injectable( {
  providedIn: 'root'
})

export class Dataservice {
   /**
   * The default page size to be returned
   */
  pageSize: number = 5;

   /**
   * The constructor for the Data Service
   * @param http Injecting the HttpClient to the DataService class
   */
  constructor(private http: HttpClient) {}

/**
 *fetches a collection of movies from the Movie Directory API
 * @param page the page requested
 * @returns An Observable for the collection of movies
 */
  getMovies(page: number) {
    let pageStart = (page - 1) * this.pageSize;
    let pageEnd = pageStart + this.pageSize;
    return jsonData.slice(pageStart, pageEnd);
  }

/**
 * formula created to calculate the volume of movies per page
 * @returns a set volume of movies per page
 */
  getLastPageNumber() {
    return Math.ceil(jsonData.length / this.pageSize);
  }

/**
 *Fetch a single movie from the Movie Directory API
 * @param id the ID indentifies the movie requested
 * @returns An Observable for the movie object
 */
  getMovie (id: any) {
    let dataToReturn: any[] = [];
    jsonData.forEach( function(movie){
        if (movie['_id']['$oid'] == id) {
            dataToReturn.push(movie)
      }
    })
    return dataToReturn
  }

/**
 *use of an external API to generate dummy text data
 * @param paragraphs uses an API to form a string of text for a movie description
 * @returns a paragraph of text from Lorem Ipsum API
 */
  getLoremIpsum(paragraphs: number): Observable<any> {

    let API_key = 'o1LlHDrpMx7YiXiE9hXwwQ==mrxycYUPUQJYXbj0';
    return this.http.get<any>(
      'https://api.api-ninjas.com/v1/loremipsum?paragraphs=' + paragraphs,
      { headers: { 'X-Api-Key': API_key}}

  }
/**
 * function that uses the API call to form a dummy description for each individual Movie in the directory
 */
  populateReviews() {
    let loremIpsum = <String>"";
    let dummyReview = <any>{};

    this.getLoremIpsum(1).subscribe( (response: any) => {
      loremIpsum = response.text;

      jsonData.forEach( function(movie){
        let numReviews = Math.floor(Math.random() * 10);
        for(var i=0; i< numReviews; i++) {
          let textSize = Math.floor(Math.random() * 290 + 10);
          let textStart = Math.floor(Math.random() * (loremIpsum.length - textSize));
          dummyReview = {
            'username' : 'User' + Math.floor(Math.random() * 999 + 1),
            'comment' : loremIpsum.slice(textStart, textStart + textSize),
            'stars' : Math.floor(Math.random() * 5) + 1
          };
          movie['reviews'].push(dummyReview);
        }
      })
    })

  }
/**
 *
 * @param id a new ID that will be linked with the new review
 * @param review the new review to be added to the movies page
 */
  postReview(id: any, review: any) {
      let newReview = {
          '_id' : review.id,
          'username' : review.username,
          'comment' : review.comment,
          'stars' : review.stars
      };
      jsonData.forEach( function(movie) {
          if (movie['_id']['$oid'] == id )
              movie['reviews'].push( newReview );
      });
    }

}
